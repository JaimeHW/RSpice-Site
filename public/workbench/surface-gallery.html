module sensor_bridge(inp, inn, out, vdd, vss);
  inout inp, inn, out, vdd, vss;
  electrical inp, inn, out, vdd, vss;
 
  parameter real gain = 40.0 from (0.0:inf);
  parameter real vos_sigma = 12u from [0.0:inf);
  parameter real rout = 10 from (0.0:inf);
  parameter real trise = 50n from (0.0:inf);
  real target, offset;
 
  analog begin
    @(initial_step) begin
      offset = $rdist_normal(seed, 0.0, vos_sigma);
    end
    target = gain * (V(inp, inn) + offset);
    target = min(V(vdd) - 0.12, target);
    target = max(V(vss) + 0.12, target);
    V(out) <+ transition(target, 0, trise);
    I(out) <+ V(out) / rout;
 
    // Limit event frequency in long transient runs
    @(timer(0, sample_period)) $bound_step(sample_period / 10);
  end
endmodule
